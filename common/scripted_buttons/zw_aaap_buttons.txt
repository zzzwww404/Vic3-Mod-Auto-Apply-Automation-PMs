# automation basic

zw_auto_pm_automation_basic_button = {
	name = "zw_auto_pm_automation_basic_button"
	desc = "zw_auto_pm_automation_basic_button_desc"
	selected = { has_variable = zw_var_auto_pm_automation_basic }
	effect = {
		custom_tooltip = zw_auto_pm_automation_basic_button_effect
		if = {
			limit = {
				NOT = { has_variable = zw_var_auto_pm_automation_basic }
			}
			set_variable = zw_var_auto_pm_automation_basic
		}
	}
}

zw_auto_pm_automation_basic_undo_button = {
	name = "zw_auto_pm_automation_basic_undo_button"
	desc = "zw_auto_pm_automation_basic_undo_button_desc"
	selected = {
		NOT = { has_variable = zw_var_auto_pm_automation_basic }
	}
	effect = {
		custom_tooltip = zw_auto_pm_automation_basic_undo_button_effect
		if = {
			limit = { has_variable = zw_var_auto_pm_automation_basic }
			remove_variable =zw_var_auto_pm_automation_basic
		}
		if = {
			limit = { has_variable = zw_var_auto_pm_automation_electric }
			remove_variable =zw_var_auto_pm_automation_electric
		}
		if = {
			limit = { has_variable = zw_var_auto_pm_automation_oil }
			remove_variable =zw_var_auto_pm_automation_oil
		}
	}
}
