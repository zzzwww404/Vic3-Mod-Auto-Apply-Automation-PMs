	trigger_if = {
		limit = { is_building_type = building_coal_mine}
		#OR = {
			# pmg_1

				limit = { has_active_production_method = pm_rail_transport_mine }
				always = no

			trigger_else_if = {
				limit = {
					state = {
						can_activate_production_method = {
							building_type = building_coal_mine
							production_method = pm_rail_transport_mine
						}
					}
				}
				always = yes
			}
			trigger_else_if = { always = no }
		#}
